var 和 dynamic

var未赋初值时，默认值为null，赋值后数据类型可变。

var variable;
print(variable);    // null
variable = '我是字符串'
variable = 10.00

dynamic未赋初值时，默认值为null，赋值后数据类型可变，同var。

List.map()

常用情景：一个对象列表，要将列表中所有对象的某个属性提取出来进行某些操作，生成一个新的列表：

List<UserModel> userList = [UserModel(name: '张三', age: 20), UserModel(name: '李四', age: 18)];
List<String> nameList = userList.map((user) => user.name).toList();

Dart中的工厂模式

import 'package:flutter/material.dart';

class DebugButton extends StatelessWidget {
  const DebugButton({
    super.key,
    required this.text,
    this.onPressed,
    this.backgroundColor,
  });

  final String text;
  final void Function()? onPressed;
  final Color? backgroundColor;

  /// Primary按钮
  const factory DebugButton.primary({
    required String text,
    Color? backgroundColor,
    void Function()? onPressed,
  }) = _DebugPrimaryButton;

  @override
  Widget build(BuildContext context) {
    return OutlinedButton(
      onPressed: () => onPressed?.call(),
      style: OutlinedButton.styleFrom(
        padding: const EdgeInsets.symmetric(horizontal: 50, vertical: 20),
        backgroundColor: backgroundColor,
        shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(30)),
      ),
      child: Text(text),
    );
  }
}

class _DebugPrimaryButton extends DebugButton {
  const _DebugPrimaryButton({
    required super.text,
    super.backgroundColor,
    super.onPressed,
  });

  @override
  Widget build(BuildContext context) {
    return ElevatedButton(
      onPressed: () => onPressed?.call(),
      style: ElevatedButton.styleFrom(
        padding: const EdgeInsets.symmetric(horizontal: 50, vertical: 20),
        backgroundColor: backgroundColor,
        shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(30)),
      ),
      child: Text(text),
    );
  }
}
